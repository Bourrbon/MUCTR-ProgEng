@startuml
class Сотрудник {
  - имя
  - ID_сотрудника
  - помочь_клиенту()
  - оформить_заказ()
}

class Меню {
  - добавить_блюдо()
  - обновить_меню()
}

class Аналитика_продаж {
  - требуемый_период()
  - сумма_прогноза()
}

class Клиент {
  - имя
  - ID_клиента
  - контактные_данные
  - сделать_заказ()
  - оплатить_заказ()
}

class Заказ {
  - ID_заказа
  - ID_клиента
  - остав
  - стоимость
  - статус
  - подтвердить_заказ()
  - отменить_заказ()
}

class Курьер {
  - имя
  - ID_курьера
  - транспорт
  - доставить_заказ()
}

class Система {
  - получить_данные_заказа()
  - назначить_курьера()
}

class Доставка {
  - статус
  - дата/время
  - ID_курьера
  - подтвердить_доставку()
  - отменить_доставку()
}

class Чек {
  - подготовить()
  - отобразить()
}

Система --> Меню
